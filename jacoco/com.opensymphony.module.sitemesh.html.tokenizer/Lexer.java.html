          case 14: break;
          case 4:
            { return Parser.WHITESPACE;
            }
          // fall through
          case 15: break;
          case 5:
            { reportError("Illegal character <"+ yytext() +">", line(), column()); return Parser.TEXT;
            }
          // fall through
          case 16: break;
          case 6:
            { return Parser.SLASH;
            }
          // fall through
          case 17: break;
          case 7:
            { return Parser.EQUALS;
            }
          // fall through
          case 18: break;
          case 8:
            { yybegin(YYINITIAL); return Parser.GT;
            }
          // fall through
          case 19: break;
          case 9:
            { return Parser.QUOTED;
            }
          // fall through
          case 20: break;
          case 10:
            { yybegin(ELEMENT); return Parser.LT_CLOSE_MAGIC_COMMENT;
            }
          // fall through
          case 21: break;
          case 11:
            { yybegin(ELEMENT); return Parser.LT_OPEN_MAGIC_COMMENT;
            }
          // fall through
          case 22: break;
          default:
            zzScanError(ZZ_NO_MATCH);
        }
      }
    }
  }
