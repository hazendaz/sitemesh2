14  package com.opensymphony.module.sitemesh.multipass;
15  
16  import com.opensymphony.module.sitemesh.DefaultSitemeshBuffer;
17  import com.opensymphony.module.sitemesh.Page;
18  import com.opensymphony.module.sitemesh.PageParser;
19  
20  import java.io.IOException;
21  
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  /**
26   * The Class DivExtractingPageParserTest.
27   *
28   * @author Joe Walnes
29   */
30  public class DivExtractingPageParserTest {
31  
32      /**
33       * Test replaces top level divs with place holders.
34       *
35       * @throws IOException
36       *             Signals that an I/O exception has occurred.
37       */
38      @Test
39      public void testReplacesTopLevelDivsWithPlaceHolders() throws IOException {
40          String input = "" + "<html>\n" + "  <head><title>Title</title></head>\n" + "  <body>\n"
41                  + "    <div id='one'>Hello</div>\n" + "    Blah\n"
42                  + "    <div id='two'>World<br><div id=inner>Great</div></div>\n" + "    <div>Bye</div>\n"
43                  + "  </body>\n" + "</html>";
44  
45          PageParser parser = new DivExtractingPageParser();
46          Page page = parser.parse(new DefaultSitemeshBuffer(input.toCharArray()));
47  
48          String expectedBody = "" + "    <sitemesh:multipass id=\"div.one\"/>\n" + "    Blah\n"
49                  + "    <sitemesh:multipass id=\"div.two\"/>\n" + "    <div>Bye</div>\n";
50          Assert.assertEquals("Title", page.getTitle());
51          Assert.assertEquals(expectedBody.trim(), page.getBody().trim());
52          Assert.assertEquals("<div id='one'>Hello</div>", page.getProperty("div.one"));
53          Assert.assertEquals("<div id='two'>World<br><div id=inner>Great</div></div>", page.getProperty("div.two"));
54      }
55  
56      /**
57       * Test extract attributes.
58       *
59       * @throws IOException
60       *             Signals that an I/O exception has occurred.
61       */
62      @Test
63      public void testExtractAttributes() throws IOException {
64          String input = "" + "<html>\n" + "  <head><title>Title</title></head>\n" + "  <body>\n"
65                  + "    <div id='one' class='c_one' align='center'>Hello</div>\n" + "    Blah\n"
66                  + "    <div id='two'>World<br><div id=inner>Great</div></div>\n" + "    <div>Bye</div>\n"
67                  + "  </body>\n" + "</html>";
68  
69          PageParser parser = new DivExtractingPageParser();
70          Page page = parser.parse(input.toCharArray());
71  
72          Assert.assertEquals("c_one", page.getProperty("div.one.class"));
73          Assert.assertEquals("center", page.getProperty("div.one.align"));
74          Assert.assertEquals("two", page.getProperty("div.two.id"));
75      }
76  }
