14  package com.opensymphony.module.sitemesh.mapper;
15  
16  import java.io.BufferedWriter;
17  import java.io.File;
18  import java.nio.charset.StandardCharsets;
19  import java.nio.file.Files;
20  
21  import org.junit.After;
22  import org.junit.Assert;
23  import org.junit.Before;
24  import org.junit.Test;
25  
26  /**
27   * The Class ConfigLoaderTest.
28   */
29  public class ConfigLoaderTest {
30  
31      /** The config loader. */
32      private ConfigLoader configLoader;
33  
34      /** The temp config file. */
35      private File tempConfigFile;
36  
37      /**
38       * Setup the unit tests.
39       *
40       * @throws Exception
41       *             the exception
42       */
43      @Before
44      public void setUp() throws Exception {
45          // create temp file
46          tempConfigFile = File.createTempFile("decorators-test", ".xml");
47          tempConfigFile.deleteOnExit();
48  
49          // write to temp file
50          BufferedWriter out = Files.newBufferedWriter(tempConfigFile.toPath(), StandardCharsets.UTF_8);
51          out.write("<decorators defaultdir=\"/decorators\">");
52  
53          // new format test decorators
54  
55          out.write("   <decorator name=\"default\" page=\"default.jsp\">");
56          out.write("       <pattern>/info/*</pattern>");
57          out.write("       <url-pattern>   ");
58          out.write("           /test/*");
59          out.write("       </url-pattern>");
60          out.write("       <url-pattern> </url-pattern>");
61          out.write("       <url-pattern></url-pattern>");
62          out.write("   </decorator>");
63  
64          out.write("   <decorator name=\"other\" page=\"/other.jsp\">");
65          out.write("       <pattern>/other/*</pattern>");
66          out.write("   </decorator>");
67  
68          out.write("   <decorator name=\"uri\" page=\"uri.jsp\" webapp=\"someapp\">");
69          out.write("       <pattern>/uri/*</pattern>");
70          out.write("   </decorator>");
71  
72          out.write("   <decorator name=\"rolebased\" page=\"rolebased.jsp\" role=\"developer\">");
73          out.write("       <pattern>/rolebased/*</pattern>");
74          out.write("   </decorator>");
75  
76          // old format test decorator
77  
78          out.write("   <decorator>");
79          out.write("       <decorator-name>old</decorator-name>");
80          out.write("       <resource>/decorators/old.jsp</resource>");
81          out.write("   </decorator>");
82  
83          out.write("   <decorator-mapping>");
84          out.write("       <decorator-name>old</decorator-name>");
85          out.write("       <url-pattern>/old/*</url-pattern>");
86          out.write("   </decorator-mapping>");
87  
88          out.write("   <decorator-mapping>");
89          out.write("       <decorator-name>old2</decorator-name>");
90          out.write("       <url-pattern></url-pattern>");
91          out.write("   </decorator-mapping>");
92  
93          out.write("</decorators>");
94          out.close();
95          configLoader = new ConfigLoader(tempConfigFile);
96      }
97  
98      /**
99       * Cleanup the unit tests.
100      *
101      * @throws Exception
102      *             the exception
103      */
104     @After
105     public void tearDown() throws Exception {
106         if (tempConfigFile != null) {
107             tempConfigFile.delete();
108         }
109         configLoader = null;
110     }
111 
112     /**
113      * Test mapped names.
114      *
115      * @throws Exception
116      *             the exception
117      */
118     @Test
119     public void testMappedNames() throws Exception {
120         Assert.assertEquals("default", configLoader.getMappedName("/info/somepage.html"));
121         Assert.assertEquals("default", configLoader.getMappedName("/test/somepage.html"));
122 
123         Assert.assertEquals("other", configLoader.getMappedName("/other/someotherpage.html"));
124 
125         Assert.assertEquals("uri", configLoader.getMappedName("/uri/somepage.html"));
126 
127         Assert.assertEquals("rolebaseddeveloper", configLoader.getMappedName("/rolebased/someotherpage.html"));
128 
129         Assert.assertEquals("old", configLoader.getMappedName("/old/someoldpage.html"));
130     }
131 
132     /**
133      * Test decorator presence.
134      *
135      * @throws Exception
136      *             the exception
137      */
138     @Test
139     public void testDecoratorPresence() throws Exception {
140         Assert.assertNotNull(configLoader.getDecoratorByName("default"));
141         Assert.assertNotNull(configLoader.getDecoratorByName("other"));
142         Assert.assertNotNull(configLoader.getDecoratorByName("uri"));
143         Assert.assertNotNull(configLoader.getDecoratorByName("rolebaseddeveloper"));
144         Assert.assertNotNull(configLoader.getDecoratorByName("old"));
145     }
146 
147     /**
148      * Test decorators.
149      *
150      * @throws Exception
151      *             the exception
152      */
153     @Test
154     public void testDecorators() throws Exception {
155         Assert.assertEquals("default", configLoader.getDecoratorByName("default").getName());
156         Assert.assertEquals("/decorators/default.jsp", configLoader.getDecoratorByName("default").getPage());
157         Assert.assertNull(configLoader.getDecoratorByName("default").getURIPath());
158         Assert.assertNull(configLoader.getDecoratorByName("default").getRole());
159 
160         Assert.assertEquals("other", configLoader.getDecoratorByName("other").getName());
161         Assert.assertEquals("/other.jsp", configLoader.getDecoratorByName("other").getPage()); // absolute path
162         Assert.assertNull(configLoader.getDecoratorByName("other").getURIPath());
163         Assert.assertNull(configLoader.getDecoratorByName("other").getRole());
164 
165         Assert.assertEquals("uri", configLoader.getDecoratorByName("uri").getName());
166         Assert.assertEquals("/decorators/uri.jsp", configLoader.getDecoratorByName("uri").getPage());
167         Assert.assertEquals("/someapp", configLoader.getDecoratorByName("uri").getURIPath());
168         Assert.assertNull(configLoader.getDecoratorByName("uri").getRole());
169 
170         Assert.assertEquals("rolebased", configLoader.getDecoratorByName("rolebaseddeveloper").getName());
171         Assert.assertEquals("/decorators/rolebased.jsp",
172                 configLoader.getDecoratorByName("rolebaseddeveloper").getPage());
173         Assert.assertNull(configLoader.getDecoratorByName("rolebaseddeveloper").getURIPath());
174         Assert.assertEquals("developer", configLoader.getDecoratorByName("rolebaseddeveloper").getRole());
175 
176         Assert.assertEquals("old", configLoader.getDecoratorByName("old").getName());
177         Assert.assertEquals("/decorators/old.jsp", configLoader.getDecoratorByName("old").getPage());
178         Assert.assertNull(configLoader.getDecoratorByName("old").getURIPath());
179         Assert.assertNull(configLoader.getDecoratorByName("old").getRole());
180     }
181 }
