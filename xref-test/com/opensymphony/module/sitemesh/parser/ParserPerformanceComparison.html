33  /**
34   * This performance test compares the three HTML parsers performance. It downloads a large real world HTML file (at time
35   * of writing this file was 676KB and growing, this is a file that is generated using sitemesh), and then parses that
36   * with each parser 1000 times over. Between each parse it does a System.gc(), to ensure garbage left over from the last
37   * parser doesn't have to be collected during the next parsers run. And it runs each parser 3 times. This should ensure
38   * that by the third time, all major JIT has been done, and so the system will be as close to a running production
39   * server as possible.
40   */
41  public class ParserPerformanceComparison {
42  
43      /** The Constant HTML_FILE. */
44      private static final String HTML_FILE = "sitemesh-performance-test.html";
45  
46      /** The Constant HTML_URL. */
47      private static final String HTML_URL = "https://jira.atlassian.com/browse/JRA-1330";
48  
49      /** The Constant PARSE_COUNT. */
50      private static final int PARSE_COUNT = 1000;
51  
52      /**
53       * The main method.
54       *
55       * @param args
56       *            the arguments
57       *
58       * @throws Exception
59       *             the exception
60       */
61      public static void main(String... args) throws Exception {
62          // Download the file if it doesn't exist
63          File file = new File(System.getProperty("java.io.tmpdir"), HTML_FILE);
64          if (!file.exists()) {
65              System.out.println("Downloading " + HTML_URL + " to use for performance test");
66              URL url = new URL(HTML_URL);
67              try (InputStream is = url.openStream(); OutputStream os = new FileOutputStream(file)) {
68                  copy(is, os);
69              }
70          } else {
71              System.out.println("Using cached file " + file);
72          }
73          // Read the cached file into a buffer
74          CharArrayWriter writer = new CharArrayWriter();
75          try (BufferedReader reader = Files.newBufferedReader(file.toPath(), StandardCharsets.UTF_8)) {
76              copy(reader, writer);
77          }
78  
79          char[] page = writer.toCharArray();
80          // Create the parsers
81          PageParser normal = new HTMLPageParser();
82          @SuppressWarnings("deprecation")
83          PageParser fast = new FastPageParser();
84          PageParser superfast = new PartialPageParser();
85  
86          System.out.println("Amount of data: " + page.length);
87          System.gc();
88          runPerformanceTest("Normal #1", page, normal, PARSE_COUNT);
89          System.gc();
90          runPerformanceTest("Fast #1", page, fast, PARSE_COUNT);
91          System.gc();
92          runPerformanceTest("Super Fast #1", page, superfast, PARSE_COUNT);
93          System.gc();
94          runPerformanceTest("Normal #2", page, normal, PARSE_COUNT);
95          System.gc();
96          runPerformanceTest("Fast #2", page, fast, PARSE_COUNT);
97          System.gc();
98          runPerformanceTest("Super Fast #2", page, superfast, PARSE_COUNT);
99          System.gc();
100         double normalTime = runPerformanceTest("Normal #3", page, normal, PARSE_COUNT);
101         System.gc();
102         double fastTime = runPerformanceTest("Fast #3", page, fast, PARSE_COUNT);
103         System.gc();
104         double superfastTime = runPerformanceTest("Super Fast #3", page, superfast, PARSE_COUNT);
105 
106         System.out.println("\nPerformance comparison %\n========================");
107         System.out.println(String.format("%-10s%12s%12s%12s", "", "Normal", "Fast", "Super Fast"));
108         System.out.println(String.format("%-10s%12s% 11.1f%%% 11.1f%%", "Normal", "X", normalTime / fastTime * 100,
109                 normalTime / superfastTime * 100));
110         System.out.println(String.format("%-10s% 11.1f%%%12s% 11.1f%%", "Fast", fastTime / normalTime * 100, "X",
111                 fastTime / superfastTime * 100));
112         System.out.println(String.format("%-10s% 11.1f%%% 11.1f%%%12s", "Super Fast", superfastTime / normalTime * 100,
113                 superfastTime / fastTime * 100, "X"));
114     }
115 
116     /**
117      * Run performance test.
118      *
119      * @param name
120      *            the name
121      * @param data
122      *            the data
123      * @param parser
124      *            the parser
125      * @param times
126      *            the times
127      *
128      * @return the long
129      *
130      * @throws Exception
131      *             the exception
132      */
133     public static long runPerformanceTest(String name, char[] data, PageParser parser, int times) throws Exception {
134         Writer writer = new NullWriter();
135         long start = System.currentTimeMillis();
136         for (int i = 0; i < times; i++) {
137             Page page = parser.parse(new DefaultSitemeshBuffer(data));
138             page.writeBody(writer);
139         }
140         long finish = System.currentTimeMillis();
141         long time = finish - start;
142         System.out.println(name + " total: " + time + "ms");
143         System.out.println(name + " average: " + (double) time / (double) times + "ms");
144         return time;
145     }
146 
147     /**
148      * The Class NullWriter.
149      */
150     private static class NullWriter extends Writer {
151         @Override
152         public void write(char[] cbuf, int off, int len) throws IOException {
153             // do nothnig
154         }
155 
156         @Override
157         public void flush() throws IOException {
158         }
159 
160         @Override
161         public void close() throws IOException {
162         }
163     }
164 
165     /**
166      * Copy.
167      *
168      * @param is
169      *            the is
170      * @param os
171      *            the os
172      *
173      * @throws IOException
174      *             Signals that an I/O exception has occurred.
175      */
176     private static void copy(InputStream is, OutputStream os) throws IOException {
177         byte[] buf = new byte[4096];
178         int length;
179         while ((length = is.read(buf)) > 0) {
180             os.write(buf, 0, length);
181         }
182     }
183 
184     /**
185      * Copy.
186      *
187      * @param reader
188      *            the reader
189      * @param writer
190      *            the writer
191      *
192      * @throws IOException
193      *             Signals that an I/O exception has occurred.
194      */
195     private static void copy(Reader reader, Writer writer) throws IOException {
196         char[] buf = new char[4096];
197         int length;
198         while ((length = reader.read(buf)) > 0) {
199             writer.write(buf, 0, length);
200         }
201     }
