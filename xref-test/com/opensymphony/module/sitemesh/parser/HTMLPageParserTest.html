52  public class HTMLPageParserTest extends TestCase {
53  
54      /** The Constant PARSER_PATTERN. */
55      private static final Pattern PARSER_PATTERN = Pattern.compile("parser\\.(.+)\\.class");
56  
57      /**
58       * This test case builds a custom suite, containing a collection of smaller suites (one for each file in
59       * src/parser-tests).
60       *
61       * @return the test
62       *
63       * @throws Exception
64       *             the exception
65       */
66      public static Test suite() throws Exception {
67          TestSuite result = new TestSuite(HTMLPageParserTest.class.getName());
68  
69          Properties props = new Properties();
70          props.load(new FileInputStream("src/parser-tests/parsers.properties"));
71  
72          Collection<String> parsers = new ArrayList<>();
73          for (String key : props.stringPropertyNames()) {
74              Matcher matcher = PARSER_PATTERN.matcher(key);
75              if (matcher.matches()) {
76                  parsers.add(matcher.group(1));
77              }
78          }
79  
80          for (String p : parsers) {
81              String name = props.getProperty("parser." + p + ".class");
82              Class<? extends PageParser> parserClass = Class.forName(name).asSubclass(PageParser.class);
83              PageParser parser = parserClass.getDeclaredConstructor().newInstance();
84  
85              String filesPath = props.getProperty("parser." + p + ".tests", "src/parser-tests");
86              List<File> files = new ArrayList<>(Arrays.asList(listParserTests(new File(filesPath))));
87              Collections.sort(files);
88  
89              TestSuite suiteForParser = new TestSuite(name);
90              for (File file : files) {
91                  TestSuite suiteForFile = new TestSuite(file.getName().replace('.', '_'));
92                  suiteForFile.addTest(new HTMLPageParserTest(parser, file, "testTitle"));
93                  suiteForFile.addTest(new HTMLPageParserTest(parser, file, "testBody"));
94                  suiteForFile.addTest(new HTMLPageParserTest(parser, file, "testHead"));
95                  suiteForFile.addTest(new HTMLPageParserTest(parser, file, "testFullPage"));
96                  suiteForFile.addTest(new HTMLPageParserTest(parser, file, "testProperties"));
97                  suiteForFile.addTest(new HTMLPageParserTest(parser, file, "testContentSanity"));
98                  suiteForParser.addTest(suiteForFile);
99              }
100             result.addTest(suiteForParser);
101         }
102 
103         return result;
104     }

136     }

138     @Override
139     protected void setUp() throws Exception {
140         super.setUp();
141         // read blocks from input file.
142         this.blocks = readBlocks(Files.newBufferedReader(file.toPath(), StandardCharsets.UTF_8));
143         // create PageParser and parse input block into HTMLPage object.
144         String input = (String) blocks.get("INPUT");
145         this.page = parser.parse(new DefaultSitemeshBuffer(input.toCharArray()));
146     }

171     /**
172      * Test head.
173      *
174      * @throws Exception
175      *             the exception
176      */
177     public void testHead() throws Exception {
178         String head;
179         if (page instanceof HTMLPage) {
180             StringWriter headWriter = new StringWriter();
181             ((HTMLPage) page).writeHead(headWriter);
182             headWriter.flush();
183             head = headWriter.toString();
184         } else {
185             head = "";
186         }
187         assertBlock("HEAD", head.toString());
188     }

203     /**
204      * Test properties.
205      *
206      * @throws Exception
207      *             the exception
208      */
209     public void testProperties() throws Exception {
210         Properties props = new Properties();
211         String propsString = (String) blocks.get("PROPERTIES");
212         ByteArrayInputStream input = new ByteArrayInputStream(propsString.trim().getBytes(Charset.forName(encoding)));
213         props.load(input);
214 
215         String[] pageKeys = page.getPropertyKeys();
216         assertEquals(file.getName() + " : Unexpected number of page properties [" + join(pageKeys) + "]", props.size(),
217                 pageKeys.length);
218 
219         for (String pageKey : pageKeys) {
220             String blockValue = props.getProperty(pageKey);
221             String pageValue = page.getProperty(pageKey);
222             assertEquals(file.getName() + ": " + pageKey, blockValue == null ? null : blockValue.trim(),
223                     pageValue == null ? null : pageValue.trim());
224         }
225     }

227     /**
228      * compare difference between using parse(char[]) and parse(char[], length).
229      *
230      * @throws Exception
231      *             the exception
232      */
233     public void testContentSanity() throws Exception {
234         String input = (String) blocks.get("INPUT");
235         final char[] chars = input.toCharArray();
236         final char[] bigChars = new char[chars.length * 2 + 10]; // make it bigger
237         System.arraycopy(chars, 0, bigChars, 0, chars.length);
238         Page bigPage = parser.parse(new DefaultSitemeshBuffer(bigChars, chars.length));
239 
240         assertEquals(bigPage.getPage(), page.getPage());
241     }

257     /**
258      * List parser tests.
259      *
260      * @param dir
261      *            the dir
262      *
263      * @return the file[]
264      *
265      * @throws IOException
266      *             Signals that an I/O exception has occurred.
267      */
268     private static File[] listParserTests(File dir) throws IOException {
269         // get list of files to ignore
270         final List<String> ignoreFileNames = new ArrayList<>();
271         String line;
272         try (LineNumberReader ignoreReader = new LineNumberReader(
273                 Files.newBufferedReader(new File(dir, "ignore.txt").toPath(), StandardCharsets.UTF_8))) {
274             while ((line = ignoreReader.readLine()) != null) {
275                 ignoreFileNames.add(line);
276             }
277         }
278         return dir.listFiles(
279                 (FilenameFilter) (currentDir, name) -> name.startsWith("test") && !ignoreFileNames.contains(name));
280     }

298     /**
299      * Read input to test and break down into blocks. See parser-tests/readme.txt
300      *
301      * @param input
302      *            the input
303      *
304      * @return the map
305      *
306      * @throws IOException
307      *             Signals that an I/O exception has occurred.
308      */
309     private Map<Object, Object> readBlocks(Reader input) throws IOException {
310         Map<Object, Object> blocks = new HashMap<>();
311         LineNumberReader reader = new LineNumberReader(input);
312         String line;
313         String blockName = null;
314         StringBuilder blockContents = null;
315         while ((line = reader.readLine()) != null) {
316             if (line.startsWith("~~~ ") && line.endsWith(" ~~~")) {
317                 if (blockName != null) {
318                     blocks.put(blockName, blockContents.toString());
319                 }
320                 blockName = line.substring(4, line.length() - 4);
321                 blockContents = new StringBuilder();
322             } else if (blockName != null) {
323                 blockContents.append(line);
324                 blockContents.append('\n');
325             }
326         }
327 
328         if (blockName != null) {
329             blocks.put(blockName, blockContents.toString());
330         }
331 
332         return blocks;
333     }
