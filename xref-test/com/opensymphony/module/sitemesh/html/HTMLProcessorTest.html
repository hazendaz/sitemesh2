14  package com.opensymphony.module.sitemesh.html;
15  
16  import com.opensymphony.module.sitemesh.SitemeshBuffer;
17  import com.opensymphony.module.sitemesh.SitemeshBufferFragment;
18  import com.opensymphony.module.sitemesh.html.rules.TagReplaceRule;
19  import com.opensymphony.module.sitemesh.html.util.StringSitemeshBuffer;
20  
21  import java.io.IOException;
22  
23  import org.junit.jupiter.api.Assertions;
24  import org.junit.jupiter.api.Test;
25  
26  /**
27   * The Class HTMLProcessorTest.
28   */
29  class HTMLProcessorTest {
30  
31      /** The body. */
32      private SitemeshBufferFragment.Builder body;
33  
34      /**
35       * Creates the processor.
36       *
37       * @param input
38       *            the input
39       *
40       * @return the HTML processor
41       */
42      private HTMLProcessor createProcessor(String input) {
43          SitemeshBuffer buffer = new StringSitemeshBuffer(input);
44          body = SitemeshBufferFragment.builder().setBuffer(buffer);
45          return new HTMLProcessor(buffer, body);
46      }
47  
48      /**
49       * Test creates state transition event.
50       *
51       * @throws IOException
52       *             Signals that an I/O exception has occurred.
53       */
54      @Test
55      void createsStateTransitionEvent() throws IOException {
56          String input = "<a></a>";
57          HTMLProcessor htmlProcessor = createProcessor(input);
58  
59          State defaultState = htmlProcessor.defaultState();
60  
61          final StringBuilder stateLog = new StringBuilder();
62  
63          defaultState.addListener(() -> stateLog.append("finished"));
64  
65          htmlProcessor.process();
66          Assertions.assertEquals("finished", stateLog.toString());
67      }
68  
69      /**
70       * Test supports conventional reader and writer.
71       *
72       * @throws IOException
73       *             Signals that an I/O exception has occurred.
74       */
75      @Test
76      void supportsConventionalReaderAndWriter() throws IOException {
77          HTMLProcessor processor = createProcessor("<hello><b id=\"something\">world</b></hello>");
78          processor.addRule(new TagReplaceRule("b", "strong"));
79  
80          processor.process();
81          Assertions.assertEquals("<hello><strong id=\"something\">world</strong></hello>",
82                  body.build().getStringContent());
83      }
84  
85      /**
86       * Test allows rules to modify attributes.
87       *
88       * @throws IOException
89       *             Signals that an I/O exception has occurred.
90       */
91      @Test
92      void allowsRulesToModifyAttributes() throws IOException {
93          HTMLProcessor processor = createProcessor("<hello><a href=\"modify-me\">world</a></hello>");
94          processor.addRule(new BasicRule("a") {
95              @Override
96              public void process(Tag tag) {
97                  currentBuffer().delete(tag.getPosition(), tag.getLength());
98                  CustomTag customTag = new CustomTag(tag);
99                  String href = customTag.getAttributeValue("href", false);
100                 if (href != null) {
101                     href = href.toUpperCase();
102                     customTag.setAttributeValue("href", true, href);
103                 }
104                 customTag.writeTo(currentBuffer(), tag.getPosition());
105             }
106         });
107 
108         processor.process();
109         Assertions.assertEquals("<hello><a href=\"MODIFY-ME\">world</a></hello>", body.build().getStringContent());
110     }
111 
112     /**
113      * Test supports chained filtering of text content.
114      *
115      * @throws IOException
116      *             Signals that an I/O exception has occurred.
117      */
118     @Test
119     void supportsChainedFilteringOfTextContent() throws IOException {
120         HTMLProcessor processor = createProcessor("<hello>world</hello>");
121         processor.addTextFilter(text -> text.toUpperCase());
122         processor.addTextFilter(text -> text.replace('O', 'o'));
123 
124         processor.process();
125         Assertions.assertEquals("<HELLo>WoRLD</HELLo>", body.build().getStringContent());
126     }
127 
128     /**
129      * Test supports text filters for specific states.
130      *
131      * @throws IOException
132      *             Signals that an I/O exception has occurred.
133      */
134     @Test
135     void supportsTextFiltersForSpecificStates() throws IOException {
136         HTMLProcessor processor = createProcessor("la la<br> la la <capitalism>laaaa<br> laaaa</capitalism> la la");
137         State capsState = new State();
138         processor.addRule(new StateTransitionRule("capitalism", capsState, true));
139 
140         capsState.addTextFilter(text -> text.toUpperCase());
141 
142         processor.process();
143         Assertions.assertEquals("la la<br> la la <capitalism>LAAAA<BR> LAAAA</capitalism> la la",
144                 body.build().getStringContent());
145     }
146 
147     /**
148      * Test can add attributes to custom tag.
149      *
150      * @throws IOException
151      *             Signals that an I/O exception has occurred.
152      */
153     @Test
154     void canAddAttributesToCustomTag() throws IOException {
155         String html = "<h1>Headline</h1>";
156         HTMLProcessor htmlProcessor = createProcessor(html);
157         htmlProcessor.addRule(new BasicRule() {
158             @Override
159             public boolean shouldProcess(String tag) {
160                 return tag.equalsIgnoreCase("h1");
161             }
162 
163             @Override
164             public void process(Tag tag) {
165                 if (tag.getType() == Tag.OPEN) {
166                     currentBuffer().delete(tag.getPosition(), tag.getLength());
167                     CustomTag ctag = new CustomTag(tag);
168                     ctag.addAttribute("class", "y");
169                     Assertions.assertEquals(1, ctag.getAttributeCount());
170                     ctag.writeTo(currentBuffer(), tag.getPosition());
171                 }
172             }
173         });
174         htmlProcessor.process();
175         Assertions.assertEquals("<h1 class=\"y\">Headline</h1>", body.build().getStringContent());
176     }
177 }
