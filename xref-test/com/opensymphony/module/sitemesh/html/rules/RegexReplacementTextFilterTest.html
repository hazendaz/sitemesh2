14  package com.opensymphony.module.sitemesh.html.rules;
15  
16  import com.opensymphony.module.sitemesh.DefaultSitemeshBuffer;
17  import com.opensymphony.module.sitemesh.SitemeshBuffer;
18  import com.opensymphony.module.sitemesh.SitemeshBufferFragment;
19  import com.opensymphony.module.sitemesh.html.HTMLProcessor;
20  
21  import java.io.IOException;
22  
23  import org.junit.jupiter.api.Assertions;
24  import org.junit.jupiter.api.Test;
25  
26  /**
27   * The Class RegexReplacementTextFilterTest.
28   */
29  class RegexReplacementTextFilterTest {
30  
31      /** The body. */
32      private SitemeshBufferFragment.Builder body;
33  
34      /**
35       * Creates the processor.
36       *
37       * @param input
38       *            the input
39       *
40       * @return the HTML processor
41       */
42      private HTMLProcessor createProcessor(String input) {
43          SitemeshBuffer buffer = new DefaultSitemeshBuffer(input.toCharArray());
44          body = SitemeshBufferFragment.builder().setBuffer(buffer);
45          return new HTMLProcessor(buffer, body);
46      }
47  
48      /**
49       * Test replaces text content matched by regular expression.
50       *
51       * @throws IOException
52       *             Signals that an I/O exception has occurred.
53       */
54      @Test
55      void replacesTextContentMatchedByRegularExpression() throws IOException {
56          HTMLProcessor processor = createProcessor("<hello>Today is DATE so hi</hello>");
57          processor.addTextFilter(new RegexReplacementTextFilter("DATE", "1-jan-2009"));
58  
59          processor.process();
60          Assertions.assertEquals("<hello>Today is 1-jan-2009 so hi</hello>", body.build().getStringContent());
61      }
62  
63      /**
64       * Test allows matched group to be used in subsitution.
65       *
66       * @throws IOException
67       *             Signals that an I/O exception has occurred.
68       */
69      @Test
70      void allowsMatchedGroupToBeUsedInSubsitution() throws IOException {
71          HTMLProcessor processor = createProcessor("<hello>I think JIRA:SIM-1234 is the way forward</hello>");
72          processor.addTextFilter(new RegexReplacementTextFilter("JIRA:([A-Z]+\\-[0-9]+)",
73                  "<a href='http://jira.opensymhony.com/browse/$1'>$1</a>"));
74  
75          processor.process();
76          Assertions.assertEquals(
77                  "<hello>I think <a href='http://jira.opensymhony.com/browse/SIM-1234'>SIM-1234</a> is the way forward</hello>",
78                  body.build().getStringContent());
79      }
80  
81  }
