36  public class DebugResponseWrapper extends HttpServletResponseWrapper {

38      /** The Logger. */
39      private static final Logger logger = LoggerFactory.getLogger(DebugResponseWrapper.class);
40  
41      /** The last count. */
42      private static int lastCount;
43  
44      /** The count. */
45      private int count;
46  
47      /**
48       * Instantiates a new debug response wrapper.
49       *
50       * @param response
51       *            the response
52       */
53      public DebugResponseWrapper(HttpServletResponse response) {
54          super(response);
55          if (enabled()) {
56              lastCount++;
57              count = lastCount;
58              debug("<CONSTRUCT>", null, null);
59          }
60      }
61

66          }
67          super.addCookie(cookie);
68      }

276     /**
277      * Debug.
278      *
279      * @param methodName
280      *            the method name
281      * @param arg1
282      *            the arg 1
283      * @param arg2
284      *            the arg 2
285      */
286     private void debug(String methodName, String arg1, String arg2) {
287         StringBuilder s = new StringBuilder();
288         s.append("[debug ");
289         s.append(count);
290         s.append("] ");
291         s.append(methodName);
292         s.append("()");
293         if (arg1 != null) {
294             s.append(" : '");
295             s.append(arg1);
296             s.append("'");
297         }
298         if (arg2 != null) {
299             s.append(" = '");
300             s.append(arg2);
301             s.append("'");
302         }
303         logger.info("{}", s);
304     }
