17  package com.opensymphony.module.sitemesh.filter;
18  
19  import javax.servlet.RequestDispatcher;
20  import javax.servlet.http.HttpServletRequest;
21  import javax.servlet.http.HttpServletRequestWrapper;
22  
23  /**
24   * Will wrap a request for the {@link RequestDispatcherWrapper}.
25   *
26   * @author <a href="mailto:joeo@enigmastation.com">Joseph B. Ottinger</a>
27   */
28  public class PageRequestWrapper extends HttpServletRequestWrapper {
29  
30      // TODO - pull this from a config file
31      /** The Constant SUPPRESS_IF_MODIFIED_HEADER. */
32      private static final boolean SUPPRESS_IF_MODIFIED_HEADER = true;
33  
34      /**
35       * Instantiates a new page request wrapper.
36       *
37       * @param request
38       *            the request
39       */
40      public PageRequestWrapper(HttpServletRequest request) {
41          super(request);
42      }
43  
44      @Override
45      public RequestDispatcher getRequestDispatcher(String s) {
46          return new RequestDispatcherWrapper(super.getRequestDispatcher(s));
47      }
48  
49      @Override
50      public String getHeader(String string) {
51          // suppress 'if-modified-since' header, so that decorators can modify the response.
52          if (SUPPRESS_IF_MODIFIED_HEADER && "IF-MODIFIED-SINCE".equalsIgnoreCase(string)) {
53              return "";
54          }
55          return super.getHeader(string);
56      }
57  }
