23  package com.opensymphony.module.sitemesh.parser;
24  
25  import com.opensymphony.module.sitemesh.HTMLPage;
26  import com.opensymphony.module.sitemesh.SitemeshBuffer;
27  
28  import java.io.IOException;
29  import java.io.Writer;
30  
31  /**
32   * Abstract implementation of {@link com.opensymphony.module.sitemesh.HTMLPage}.
33   * <p>
34   * Adds to {@link com.opensymphony.module.sitemesh.parser.AbstractPage} some HTML methods. To implement, follow
35   * guidelines of super-class, and implement the 2 abstract methods states below.
36   *
37   * @author <a href="joe@truemesh.com">Joe Walnes</a>
38   *
39   * @see com.opensymphony.module.sitemesh.parser.AbstractPage
40   * @see com.opensymphony.module.sitemesh.HTMLPage
41   */
42  public abstract class AbstractHTMLPage extends AbstractPage implements HTMLPage {
43  
44      /**
45       * Instantiates a new abstract HTML page.
46       *
47       * @param sitemeshBuffer
48       *            the sitemesh buffer
49       */
50      protected AbstractHTMLPage(SitemeshBuffer sitemeshBuffer) {
51          super(sitemeshBuffer);
52      }
53  
54      /**
55       * Write data of html <code>&lt;head&gt;</code> tag.
56       * <p>
57       * Must be implemented. Data written should not actually contain the head tags, but all the data in between.
58       */
59      @Override
60      public abstract void writeHead(Writer out) throws IOException;
61  
62      @Override
63      public boolean isFrameSet() {
64          return isPropertySet("frameset") && getProperty("frameset").equalsIgnoreCase("true");
65      }
66  
67      @Override
68      public void setFrameSet(boolean frameset) {
69          if (frameset) {
70              addProperty("frameset", "true");
71          } else if (isPropertySet("frameset")) {
72              addProperty("frameset", "false");
73          }
74      }
75  }
