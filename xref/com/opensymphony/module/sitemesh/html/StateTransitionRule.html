5   package com.opensymphony.module.sitemesh.html;
6   
7   /**
8    * The Class StateTransitionRule.
9    */
10  public class StateTransitionRule extends BasicRule {
11  
12      /** The new state. */
13      private final State newState;
14  
15      /** The last state. */
16      private State lastState;
17  
18      /**
19       * Instantiates a new state transition rule.
20       *
21       * @param tagName
22       *            the tag name
23       * @param newState
24       *            the new state
25       */
26      public StateTransitionRule(String tagName, State newState) {
27          this(tagName, newState, true);
28      }
29  
30      /**
31       * Instantiates a new state transition rule.
32       *
33       * @param tagName
34       *            the tag name
35       * @param newState
36       *            the new state
37       * @param writeEnclosingTag
38       *            the write enclosing tag
39       */
40      public StateTransitionRule(String tagName, State newState, boolean writeEnclosingTag) {
41          super(tagName);
42          this.newState = newState;
43      }
44  
45      @Override
46      public void process(Tag tag) {
47          if (tag.getType() == Tag.OPEN) {
48              lastState = context.currentState();
49              context.changeState(newState);
50              newState.addRule(this);
51          } else if (tag.getType() == Tag.CLOSE && lastState != null) {
52              context.changeState(lastState);
53              lastState = null;
54          }
55      }
56  }
