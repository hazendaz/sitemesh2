16  /**
17   * The listener interface for receiving stateChange events. The class that is interested in processing a stateChange
18   * event implements this interface, and the object created with that class is registered with a component using the
19   * component's <code>addStateChangeListener</code> method. When the stateChange event occurs, that object's appropriate
20   * method is invoked.
21   */
22  public interface StateChangeListener {
23  
24      /**
25       * State finished.
26       */
27      void stateFinished();
28  }
