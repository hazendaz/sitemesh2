5   package com.opensymphony.module.sitemesh.html.rules;
6   
7   import com.opensymphony.module.sitemesh.html.BasicRule;
8   import com.opensymphony.module.sitemesh.html.Tag;
9   
10  /**
11   * The Class FramesetRule.
12   */
13  public class FramesetRule extends BasicRule {
14  
15      /** The page. */
16      private final PageBuilder page;
17  
18      /**
19       * Instantiates a new frameset rule.
20       *
21       * @param page
22       *            the page
23       */
24      public FramesetRule(PageBuilder page) {
25          super(new String[] { "frame", "frameset" });
26          this.page = page;
27      }
28  
29      @Override
30      public void process(Tag tag) {
31          context.currentBuffer().delete(tag.getPosition(), tag.getLength());
32          page.addProperty("frameset", "true");
33      }
34  
35  }
