14  package com.opensymphony.module.sitemesh.html.rules;
15  
16  import com.opensymphony.module.sitemesh.html.BasicRule;
17  import com.opensymphony.module.sitemesh.html.Tag;
18  
19  /**
20   * The Class ParameterExtractingRule.
21   */
22  public class ParameterExtractingRule extends BasicRule {
23  
24      /** The page. */
25      private final PageBuilder page;
26  
27      /**
28       * Instantiates a new parameter extracting rule.
29       *
30       * @param page
31       *            the page
32       */
33      public ParameterExtractingRule(PageBuilder page) {
34          super("parameter");
35          this.page = page;
36      }
37  
38      @Override
39      public void process(Tag tag) {
40          context.currentBuffer().delete(tag.getPosition(), tag.getLength());
41          page.addProperty("page." + tag.getAttributeValue("name", false), tag.getAttributeValue("value", false));
42      }
43  }
