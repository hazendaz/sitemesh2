5   package com.opensymphony.module.sitemesh.html.rules;
6   
7   import com.opensymphony.module.sitemesh.html.BlockExtractingRule;
8   import com.opensymphony.module.sitemesh.html.Tag;
9   
10  /**
11   * The Class ContentBlockExtractingRule.
12   */
13  public class ContentBlockExtractingRule extends BlockExtractingRule {
14  
15      /** The page. */
16      private final PageBuilder page;
17  
18      /** The content block id. */
19      private String contentBlockId;
20  
21      /**
22       * Instantiates a new content block extracting rule.
23       *
24       * @param page
25       *            the page
26       */
27      public ContentBlockExtractingRule(PageBuilder page) {
28          super(false, "content");
29          this.page = page;
30      }
31  
32      @Override
33      protected void start(Tag tag) {
34          contentBlockId = tag.getAttributeValue("tag", false);
35      }
36  
37      @Override
38      protected void end(Tag tag) {
39          page.addProperty("page." + contentBlockId, getCurrentBufferContent());
40      }
41  
42  }
