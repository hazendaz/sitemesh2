14  package com.opensymphony.module.sitemesh.html.rules;
15  
16  import com.opensymphony.module.sitemesh.html.BlockExtractingRule;
17  import com.opensymphony.module.sitemesh.html.Tag;
18  
19  /**
20   * The Class ContentBlockExtractingRule.
21   */
22  public class ContentBlockExtractingRule extends BlockExtractingRule {
23  
24      /** The page. */
25      private final PageBuilder page;
26  
27      /** The content block id. */
28      private String contentBlockId;
29  
30      /**
31       * Instantiates a new content block extracting rule.
32       *
33       * @param page
34       *            the page
35       */
36      public ContentBlockExtractingRule(PageBuilder page) {
37          super(false, "content");
38          this.page = page;
39      }
40  
41      @Override
42      protected void start(Tag tag) {
43          contentBlockId = tag.getAttributeValue("tag", false);
44      }
45  
46      @Override
47      protected void end(Tag tag) {
48          page.addProperty("page." + contentBlockId, getCurrentBufferContent());
49      }
50  
51  }
