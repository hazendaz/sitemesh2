14  package com.opensymphony.module.sitemesh.html.rules;
15  
16  import com.opensymphony.module.sitemesh.html.BlockExtractingRule;
17  import com.opensymphony.module.sitemesh.html.Tag;
18  
19  /**
20   * The Class TitleExtractingRule.
21   */
22  public class TitleExtractingRule extends BlockExtractingRule {
23  
24      /** The page. */
25      private final PageBuilder page;
26  
27      /** The seen title. */
28      private boolean seenTitle;
29  
30      /**
31       * Instantiates a new title extracting rule.
32       *
33       * @param page
34       *            the page
35       */
36      public TitleExtractingRule(PageBuilder page) {
37          super(false, "title");
38          this.page = page;
39      }
40  
41      @Override
42      protected void end(Tag tag) {
43          if (!seenTitle) {
44              page.addProperty("title", getCurrentBufferContent());
45              seenTitle = true;
46          }
47      }
48  }
