14  package com.opensymphony.module.sitemesh.html.tokenizer;
15  
16  import com.opensymphony.module.sitemesh.html.Tag;
17  import com.opensymphony.module.sitemesh.html.Text;
18  
19  /**
20   * Handler passed to {@link TagTokenizer} that will receive callbacks as 'tags' and 'text' are encountered.
21   *
22   * @author Joe Walnes
23   *
24   * @see TagTokenizer
25   */
26  public interface TokenHandler {
27  
28      /**
29       * Before attempting to parse a tag, the tokenizer will ask the handler whether the tag should be processed -
30       * avoiding additional tag parsing makes the tokenizer quicker.
31       * <p>
32       * If true is returned, the tokenizer will fully parse the tag and pass it into the
33       * {@link #tag(com.opensymphony.module.sitemesh.html.Tag)} method. If false is returned, the tokenizer will not try
34       * to parse the tag and pass it to the #{@link #text(com.opensymphony.module.sitemesh.html.Text)} method, untouched.
35       *
36       * @param name
37       *            the name
38       *
39       * @return true, if successful
40       */
41      boolean shouldProcessTag(String name);
42  
43      /**
44       * Called when tokenizer encounters an HTML tag (open, close or empty). The Tag instance passed in should not be
45       * kept beyond the scope of this method as the tokenizer will attempt to reuse it.
46       *
47       * @param tag
48       *            the tag
49       */
50      void tag(Tag tag);
51  
52      /**
53       * Called when tokenizer encounters anything other than a well-formed HTML tag. The Text object is used instead of a
54       * String to allow the String to be lazy-loaded. The Text instance passed in should not be kept beyond the scope of
55       * this method as the tokenizer will attempt to reuse it.
56       *
57       * @param text
58       *            the text
59       */
60      void text(Text text);
61  
62      /**
63       * Called when tokenizer encounters something it cannot correctly parse. Typically the parsing will continue and the
64       * unparseable will be treated as a plain text block, however this callback provides indication of this.
65       *
66       * @param message
67       *            Error message
68       * @param line
69       *            Line number in input that error occured
70       * @param column
71       *            Column number in input that error occured
72       */
73      void warning(String message, int line, int column);
74  
75  }
