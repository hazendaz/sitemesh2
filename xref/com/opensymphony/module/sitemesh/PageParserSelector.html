14  package com.opensymphony.module.sitemesh;
15  
16  /**
17   * The Interface PageParserSelector.
18   *
19   * @author Joe Walnes
20   */
21  public interface PageParserSelector {
22  
23      /**
24       * Determine whether a Page of given content-type should be parsed or not.
25       *
26       * @param contentType
27       *            the content type
28       *
29       * @return true, if successful
30       */
31      boolean shouldParsePage(String contentType);
32  
33      /**
34       * Create a PageParser suitable for the given content-type.
35       * <p>
36       * For example, if the supplied parameter is <code>text/html</code> a parser shall be returned that can parse HTML
37       * accordingly.
38       * </p>
39       * Never returns null.
40       *
41       * @param contentType
42       *            The MIME content-type of the data to be parsed
43       *
44       * @return Appropriate <code>PageParser</code> for reading data
45       */
46      PageParser getPageParser(String contentType);
47  
48  }
