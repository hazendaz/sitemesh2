5   package com.opensymphony.module.sitemesh;
6   
7   /**
8    * The Interface PageParserSelector.
9    *
10   * @author Joe Walnes
11   */
12  public interface PageParserSelector {
13  
14      /**
15       * Determine whether a Page of given content-type should be parsed or not.
16       *
17       * @param contentType
18       *            the content type
19       *
20       * @return true, if successful
21       */
22      boolean shouldParsePage(String contentType);
23  
24      /**
25       * Create a PageParser suitable for the given content-type.
26       * <p>
27       * For example, if the supplied parameter is <code>text/html</code> a parser shall be returned that can parse HTML
28       * accordingly.
29       * </p>
30       * Never returns null.
31       *
32       * @param contentType
33       *            The MIME content-type of the data to be parsed
34       *
35       * @return Appropriate <code>PageParser</code> for reading data
36       */
37      PageParser getPageParser(String contentType);
38  
39  }
