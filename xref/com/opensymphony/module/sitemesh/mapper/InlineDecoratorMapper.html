23  package com.opensymphony.module.sitemesh.mapper;
24  
25  import com.opensymphony.module.sitemesh.Decorator;
26  import com.opensymphony.module.sitemesh.Page;
27  import com.opensymphony.module.sitemesh.RequestConstants;
28  import com.opensymphony.module.sitemesh.factory.FactoryException;
29  
30  import javax.servlet.http.HttpServletRequest;
31  
32  /**
33   * The InlineDecoratorMapper is used to determine the correct Decorator when using inline decorators.
34   * <p>
35   * It will check the request attribute value defined by the key
36   * {@link com.opensymphony.module.sitemesh.RequestConstants#DECORATOR} and use the appropriate named Decorator. This is
37   * passed across from the page:applyDecorator tag.
38   *
39   * @author <a href="mailto:joe@truemesh.com">Joe Walnes</a>
40   *
41   * @see com.opensymphony.module.sitemesh.DecoratorMapper
42   */
43  public class InlineDecoratorMapper extends AbstractDecoratorMapper implements RequestConstants {
44      @Override
45      public Decorator getDecorator(HttpServletRequest request, Page page) {
46          Decorator result = null;
47          if (request.getAttribute(DECORATOR) != null) {
48              // Retrieve name of decorator to use from request
49              String decoratorName = (String) request.getAttribute(DECORATOR);
50              result = getNamedDecorator(request, decoratorName);
51              if (result == null) {
52                  throw new FactoryException("Cannot locate inline Decorator: " + decoratorName);
53              }
54          }
55          return result == null ? super.getDecorator(request, page) : result;
56      }
57  }
