98      @Override
99      public void init(Config config, Properties properties, DecoratorMapper parent) throws InstantiationException {
100         super.init(config, properties, parent);
101         decoratorName = properties.getProperty("decorator");
102     }

104     @Override
105     public Decorator getDecorator(HttpServletRequest request, Page page) {
106         Decorator result = null;
107 
108         if (decoratorName != null && isBot(request)) {
109             result = getNamedDecorator(request, decoratorName);
110         }
111 
112         return result == null ? super.getDecorator(request, page) : result;
113     }

115     /**
116      * Check if the current request came from a robot (also known as spider, crawler, ferret).
117      *
118      * @param request
119      *            the request
120      *
121      * @return true, if is bot
122      */
123     private static boolean isBot(HttpServletRequest request) {
124         if (request == null) {
125             return false;
126         }
127 
128         // force creation of a session
129         HttpSession session = request.getSession(true);
130 
131         if (Boolean.FALSE.equals(session.getAttribute(RequestConstants.ROBOT))) {
132             return false;
133         }
134         if (Boolean.TRUE.equals(session.getAttribute(RequestConstants.ROBOT))) {
135             // a key was found in the session indicating it is a robot
136             return true;
137         } else if ("robots.txt".indexOf(request.getRequestURI()) != -1) {
138             // there is a specific request for the robots.txt file, so we assume
139             // it must be a robot (only robots request robots.txt)
140 
141             // set a key in the session, so the next time we don't have to manually
142             // detect the robot again
143             session.setAttribute(RequestConstants.ROBOT, Boolean.TRUE);
144             return true;
145         } else {
146             String userAgent = request.getHeader("User-Agent");
147 
148             if (userAgent != null && userAgent.trim().length() > 2) {
149                 // first check for common user-agent headers, so that we can speed
150                 // this thing up, hopefully clever spiders will not send a fake header
151                 if (userAgent.indexOf("MSIE") != -1 || userAgent.indexOf("Gecko") != -1 // MSIE and Mozilla
152                         || userAgent.indexOf("Opera") != -1 || userAgent.indexOf("iCab") != -1 // Opera and iCab
153                         // (mac browser)
154                         || userAgent.indexOf("Konqueror") != -1 || userAgent.indexOf("KMeleon") != -1 // Konqueror
155                         // and KMeleon
156                         || userAgent.indexOf("4.7") != -1 || userAgent.indexOf("Lynx") != -1) { // NS 4.78 and Lynx
157                     // indicate this session is not a robot
158                     session.setAttribute(RequestConstants.ROBOT, Boolean.FALSE);
159                     return false;
160                 }
161 
162                 for (String botAgent : botAgents) {
163                     if (userAgent.indexOf(botAgent) != -1) {
164                         // set a key in the session, so the next time we don't have to manually
165                         // detect the robot again
166                         session.setAttribute(RequestConstants.ROBOT, Boolean.TRUE);
167                         return true;
168                     }
169                 }
170             }
171 
172             // detect the robot from the host or user-agent
173             String remoteHost = request.getRemoteHost(); // requires one DNS lookup
174 
175             // if the DNS server didn't return a hostname, getRemoteHost returns the
176             // IP address, which is ignored here (the last char is checked, because some
177             // remote hosts begin with the IP)
178             if (remoteHost != null && remoteHost.length() > 0 && remoteHost.charAt(remoteHost.length() - 1) > 64) {
179                 for (String botHost : botHosts) {
180                     if (remoteHost.indexOf(botHost) != -1) {
181                         // set a key in the session, so the next time we don't have to manually
182                         // detect the robot again
183                         session.setAttribute(RequestConstants.ROBOT, Boolean.TRUE);
184                         return true;
185                     }
186                 }
187             }
188 
189             // remote host and user agent are not in the predefined list,
190             // so it must be an unknown robot or not a robot
191 
192             // indicate this session is not a robot
193             session.setAttribute(RequestConstants.ROBOT, Boolean.FALSE);
194             return false;
195         }
196     }
