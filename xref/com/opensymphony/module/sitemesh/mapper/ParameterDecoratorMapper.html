23  package com.opensymphony.module.sitemesh.mapper;
24  
25  import com.opensymphony.module.sitemesh.Config;
26  import com.opensymphony.module.sitemesh.Decorator;
27  import com.opensymphony.module.sitemesh.DecoratorMapper;
28  import com.opensymphony.module.sitemesh.Page;
29  
30  import java.util.Properties;
31  
32  import javax.servlet.http.HttpServletRequest;
33  
34  /**
35   * The ParameterDecoratorMapper will map a suitable decorator based on request parameters.
36   * <p>
37   * The ParameterDecoratorMapper is configured via three properties.
38   * <p>
39   * <code>decorator.parameter</code> - the parameter which contains the name of the decorator which will be mapped. The
40   * default is "decorator".
41   * <p>
42   * For example if <code>decorator.parameter</code> is "foobar" then myurl.jsp?foobar=mydecorator will map to the
43   * decorator named "mydecorator".
44   * <p>
45   * You can also supply an optional 'confirmation parameter'. The decorator will only be mapped if the parameter named
46   * <code>parameter.name</code> is in the request URI and the value of that parameter is equal to the
47   * <code>parameter.value</code> property.
48   * <p>
49   * For example assuming parameter.name=confirm and parameter.value=true the URI
50   * myurl.jsp?decorator=mydecorator&amp;confirm=true will map the decorator mydecorator. where as the URIs
51   * myurl.jsp?decorator=mydecorator and myurl.jsp?decorator=mydecorator&amp;confirm=false will not return any decorator.
52   *
53   * @author <a href="mailto:mcannon@internet.com">Mike Cannon-Brookes</a>
54   *
55   * @see com.opensymphony.module.sitemesh.DecoratorMapper
56   */
57  public class ParameterDecoratorMapper extends AbstractDecoratorMapper {
58  
59      /** The param value. */
60      private String decoratorParameter = null, paramName = null, paramValue = null;
61  
62      @Override
63      public void init(Config config, Properties properties, DecoratorMapper parent) throws InstantiationException {
64          super.init(config, properties, parent);
65          decoratorParameter = properties.getProperty("decorator.parameter", "decorator");
66          paramName = properties.getProperty("parameter.name", null);
67          paramValue = properties.getProperty("parameter.value", null);
68      }
69  
70      @Override
71      public Decorator getDecorator(HttpServletRequest request, Page page) {
72          Decorator result = null;
73          String decoratorParamValue = request.getParameter(decoratorParameter);
74  
75          if ((paramName == null || paramValue.equals(request.getParameter(paramName))) && decoratorParamValue != null
76                  && !decoratorParamValue.trim().equals("")) {
77              result = getNamedDecorator(request, decoratorParamValue);
78          }
79          return result == null ? super.getDecorator(request, page) : result;
80      }
81  }
