23  package com.opensymphony.module.sitemesh.mapper;
24  
25  import com.opensymphony.module.sitemesh.Decorator;
26  
27  import java.io.File;
28  import java.net.MalformedURLException;
29  import java.net.URL;
30  
31  import javax.servlet.http.HttpServletRequest;
32  
33  /**
34   * The FileDecoratorMapper will treat the name of the decorator as a file-name to use (in the context of the web-app).
35   *
36   * @author <a href="joe@truemesh.com">Joe Walnes</a>
37   * @author <a href="mike@atlassian.com">Mike Cannon-Brookes</a>
38   *
39   * @see com.opensymphony.module.sitemesh.DecoratorMapper
40   * @see com.opensymphony.module.sitemesh.mapper.DefaultDecorator
41   */
42  public class FileDecoratorMapper extends AbstractDecoratorMapper {
43  
44      /** The path not available. */
45      private boolean pathNotAvailable = false;
46  
47      @Override
48      public Decorator getNamedDecorator(HttpServletRequest req, String name) {
49          if (pathNotAvailable || name == null) {
50              return super.getNamedDecorator(req, name);
51          }
52  
53          URL resourcePath;
54  
55          // try to locate the resource (might be an unexpanded WAR)
56          try {
57              resourcePath = config.getServletContext().getResource('/' + name);
58          } catch (MalformedURLException e) {
59              return super.getNamedDecorator(req, name);
60          }
61  
62          String filePath = config.getServletContext().getRealPath(name);
63  
64          if (filePath == null && resourcePath == null) {
65              pathNotAvailable = true;
66              return super.getNamedDecorator(req, name);
67          }
68          if (filePath != null) { // do we really need this disk file check?!
69              File file = new File(filePath);
70  
71              if (file.exists() && file.canRead() && file.isFile()) {
72                  // if filename exists with name of supplied decorator, return Decorator
73                  return new DefaultDecorator(name, name, null);
74              } else {
75                  // otherwise delegate to parent mapper.
76                  return super.getNamedDecorator(req, name);
77              }
78          } else {
79              // file path is null and resource path is not null - can't check file on disk
80              return new DefaultDecorator(name, name, null);
81          }
82      }
83  }
