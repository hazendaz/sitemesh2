23  package com.opensymphony.module.sitemesh.mapper;
24  
25  import com.opensymphony.module.sitemesh.Decorator;
26  
27  import javax.naming.Context;
28  import javax.naming.InitialContext;
29  import javax.naming.NamingException;
30  import javax.servlet.http.HttpServletRequest;
31  
32  import org.slf4j.Logger;
33  import org.slf4j.LoggerFactory;
34  
35  /**
36   * The EnvEntryDecoratorMapper allows the reference to a web-app environment entry for the decorator name, and falls
37   * back to ConfigDecoratorMapper's behavior if no matching environment entry is found.
38   * <p>
39   * In some cases, it's desirable to allow a deployer, as opposed to a developer, to specify a decorator. In a .WAR file,
40   * this can be very difficult, since decorator mappings are specified in <code>decorators.xml</code> (more or less).
41   * This mapper corrects that by allowing two types of mapping. If the decorator name is found in an
42   * <code>&lt;env-entry&gt;</code>, the entry's value is used as the decorator reference.
43   * <p>
44   * Known Issues:
45   * <ol>
46   * <li>It still uses the decorator path (from <code>decorators.xml</code>). This needs to be corrected for full
47   * functionality. If anyone has a suggestion on how...</li>
48   * </ol>
49   *
50   * @author <a href="mailto:joeo@enigmastation.com">Joseph B. Ottinger</a>
51   *
52   * @see com.opensymphony.module.sitemesh.mapper.ConfigDecoratorMapper
53   */
54  public class EnvEntryDecoratorMapper extends ConfigDecoratorMapper {
55  
56      /** The Logger. */
57      private static final Logger logger = LoggerFactory.getLogger(EnvEntryDecoratorMapper.class);
58  
59      /**
60       * Retrieves the {@link com.opensymphony.module.sitemesh.Decorator} specified by the decorator name. If it's not in
61       * the environment entries of the web application, assume it's a named decorator from <code>decorators.xml</code>.
62       */
63      @Override
64      public Decorator getNamedDecorator(HttpServletRequest request, String name) {
65          String resourceValue = getStringResource(name);
66          if (resourceValue == null) {
67              return super.getNamedDecorator(request, name);
68          }
69          return new DefaultDecorator(name, resourceValue, null);
70      }
71  
72      /**
73       * This pulls a value out of the web-app environment. If the value isn't there, returns null.
74       *
75       * @param name
76       *            the name
77       *
78       * @return the string resource
79       */
80      public static String getStringResource(String name) {
81          String value = null;
82          Context ctx = null;
83          try {
84              ctx = new InitialContext();
85              Object o = ctx.lookup("java:comp/env/" + name);
86              if (o != null) {
87                  value = o.toString();
88              }
89          } catch (NamingException e) {
90              logger.error("", e);
91          } finally {
92              try {
93                  if (ctx != null) {
94                      ctx.close();
95                  }
96              } catch (NamingException e) {
97                  logger.error("", e);
98              }
99          }
100         return value;
101     }
102 }
