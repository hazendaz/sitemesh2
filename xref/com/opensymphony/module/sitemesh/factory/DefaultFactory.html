44  /**
45   * DefaultFactory, reads configuration from the <code>sitemesh.configfile</code> init param, or
46   * <code>/WEB-INF/sitemesh.xml</code> if not specified, or uses the default configuration if <code>sitemesh.xml</code>
47   * does not exist.
48   * <p>
49   * To use the <code>sitemesh.configfile</code> parameter, add the following to your web.xml:
50   *
51   * <pre>
52   * &lt;context-param&gt;
53   *      &lt;param-name&gt;sitemesh.configfile&lt;/param-name&gt;
54   *      &lt;param-value&gt;/WEB-INF/etc/sitemesh.xml&lt;/param-value&gt;
55   *  &lt;/context-param&gt;
56   * </pre>
57   *
58   * @author <a href="mailto:joe@truemesh.com">Joe Walnes</a>
59   * @author <a href="mailto:pathos@pandora.be">Mathias Bogaert</a>
60   */
61  public class DefaultFactory extends BaseFactory {

64      String configFileName;

66      /** The Constant DEFAULT_CONFIG_FILENAME. */
67      private static final String DEFAULT_CONFIG_FILENAME = "/WEB-INF/sitemesh.xml";

69      /** The config file. */
70      File configFile;

73      long configLastModified;
74  
75      /** The config last check. */
76      private long configLastCheck = 0L;
77  
78      /** The config check millis. */
79      public static long configCheckMillis = 3000L;
80  
81      /** The config props. */
82      Map<Object, Object> configProps = new HashMap<>();

85      String excludesFileName;

88      File excludesFile;

90      /**
91       * Instantiates a new default factory.
92       *
93       * @param config
94       *            the config
95       */
96      public DefaultFactory(Config config) {
97          super(config);
98  
99          configFileName = config.getServletContext().getInitParameter("sitemesh.configfile");
100         if (configFileName == null) {
101             configFileName = DEFAULT_CONFIG_FILENAME;
102         }
103 
104         // configFilePath is null if loaded from war file
105         String initParamConfigFile = config.getConfigFile();
106         if (initParamConfigFile != null) {
107             configFileName = initParamConfigFile;
108         }
109 
110         String configFilePath = config.getServletContext().getRealPath(configFileName);
111 
112         if (configFilePath != null) { // disable config auto reloading for .war files
113             configFile = new File(configFilePath);
114         }
115 
116         loadConfig();
117     }

119     /** Load configuration from file. */
120     private synchronized void loadConfig() {
121         try {
122             // Load and parse the sitemesh.xml file
123             Element root = loadSitemeshXML();
124 
125             NodeList sections = root.getChildNodes();
126             // Loop through child elements of root node
127             for (int i = 0; i < sections.getLength(); i++) {
128                 if (sections.item(i) instanceof Element) {
129                     Element curr = (Element) sections.item(i);
130                     NodeList children = curr.getChildNodes();
131 
132                     if ("config-refresh".equalsIgnoreCase(curr.getTagName())) {
133                         String seconds = curr.getAttribute("seconds");
134                         configCheckMillis = Long.parseLong(seconds) * 1000L;
135                     } else if ("property".equalsIgnoreCase(curr.getTagName())) {
136                         String name = curr.getAttribute("name");
137                         String value = curr.getAttribute("value");
138                         if (!"".equals(name) && !"".equals(value)) {
139                             configProps.put("${" + name + "}", value);
140                         }
141                     } else if ("page-parsers".equalsIgnoreCase(curr.getTagName())) {
142                         // handle <page-parsers>
143                         loadPageParsers(children);
144                     } else if ("decorator-mappers".equalsIgnoreCase(curr.getTagName())) {
145                         // handle <decorator-mappers>
146                         loadDecoratorMappers(children);
147                     } else if ("excludes".equalsIgnoreCase(curr.getTagName())) {
148                         // handle <excludes>
149                         String fileName = replaceProperties(curr.getAttribute("file"));
150                         if (!"".equals(fileName)) {
151                             excludesFileName = fileName;
152                             loadExcludes();
153                         }
154                     }
155                 }
156             }
157         } catch (ParserConfigurationException e) {
158             throw new FactoryException("Could not get XML parser", e);
159         } catch (IOException e) {
160             throw new FactoryException("Could not read config file : " + configFileName, e);
161         } catch (SAXException e) {
162             throw new FactoryException("Could not parse config file : " + configFileName, e);
163         }
164     }

166     /**
167      * Load sitemesh XML.
168      *
169      * @return the element
170      *
171      * @throws ParserConfigurationException
172      *             the parser configuration exception
173      * @throws IOException
174      *             Signals that an I/O exception has occurred.
175      * @throws SAXException
176      *             the SAX exception
177      */
178     private Element loadSitemeshXML() throws ParserConfigurationException, IOException, SAXException {
179         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
180         factory.setFeature("http://apache.org/xml/features/disallow-doctype-decl", true);
181         DocumentBuilder builder = factory.newDocumentBuilder();
182 
183         InputStream is = null;
184 
185         if (configFile == null) {
186             is = config.getServletContext().getResourceAsStream(configFileName);
187         } else if (configFile.exists() && configFile.canRead()) {
188             is = configFile.toURI().toURL().openStream();
189         }
190 
191         // load the default sitemesh configuration
192         if (is == null) {
193             is = getClass().getClassLoader()
194                     .getResourceAsStream("com/opensymphony/module/sitemesh/factory/sitemesh-default.xml");
195         }
196 
197         // load the default sitemesh configuration using another classloader
198         if (is == null) {
199             is = Thread.currentThread().getContextClassLoader()
200                     .getResourceAsStream("com/opensymphony/module/sitemesh/factory/sitemesh-default.xml");
201         }
202 
203         if (is == null) {
204             throw new IllegalStateException("Cannot load default configuration from jar");
205         }
206 
207         if (configFile != null) {
208             configLastModified = configFile.lastModified();
209         }
210 
211         Document doc = builder.parse(is);
212         Element root = doc.getDocumentElement();
213         // Verify root element
214         if (!"sitemesh".equalsIgnoreCase(root.getTagName())) {
215             throw new FactoryException("Root element of sitemesh configuration file not <sitemesh>", null);
216         }
217         return root;
218     }

220     /**
221      * Load excludes.
222      *
223      * @throws ParserConfigurationException
224      *             the parser configuration exception
225      * @throws IOException
226      *             Signals that an I/O exception has occurred.
227      * @throws SAXException
228      *             the SAX exception
229      */
230     private void loadExcludes() throws ParserConfigurationException, IOException, SAXException {
231         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
232         factory.setFeature("http://apache.org/xml/features/disallow-doctype-decl", true);
233         DocumentBuilder builder = factory.newDocumentBuilder();
234 
235         InputStream is = null;
236 
237         if (excludesFile == null) {
238             is = config.getServletContext().getResourceAsStream(excludesFileName);
239             if (is == null) {
240                 String excludesFilePath = config.getServletContext().getRealPath(excludesFileName);// getResourceAsStream
241                                                                                                    // does not work on
242                                                                                                    // weblogic 10.3.5
243                 if (excludesFilePath != null) {
244                     excludesFile = new File(excludesFilePath);
245                     is = excludesFile.toURI().toURL().openStream();
246                 }
247             }
248         } else if (excludesFile.exists() && excludesFile.canRead()) {
249             is = excludesFile.toURI().toURL().openStream();
250         }
251 
252         if (is == null) {
253             throw new IllegalStateException("Cannot load excludes configuration file \"" + excludesFileName
254                     + "\" as specified in \"sitemesh.xml\" or \"sitemesh-default.xml\"");
255         }
256 
257         Document document = builder.parse(is);
258         Element root = document.getDocumentElement();
259         NodeList sections = root.getChildNodes();
260 
261         // Loop through child elements of root node looking for the <excludes> block
262         for (int i = 0; i < sections.getLength(); i++) {
263             if (sections.item(i) instanceof Element) {
264                 Element curr = (Element) sections.item(i);
265                 if ("excludes".equalsIgnoreCase(curr.getTagName())) {
266                     loadExcludeUrls(curr.getChildNodes());
267                 }
268             }
269         }
270     }

272     /**
273      * Loop through children of 'page-parsers' element and add all 'parser' mappings.
274      *
275      * @param nodes
276      *            the nodes
277      */
278     private void loadPageParsers(NodeList nodes) {
279         clearParserMappings();
280         for (int i = 0; i < nodes.getLength(); i++) {
281             if (nodes.item(i) instanceof Element) {
282                 Element curr = (Element) nodes.item(i);
283 
284                 if ("parser".equalsIgnoreCase(curr.getTagName())) {
285                     String className = curr.getAttribute("class");
286                     String contentType = curr.getAttribute("content-type");
287                     mapParser(contentType, className);
288                 }
289             }
290         }
291     }

293     /**
294      * Load decorator mappers.
295      *
296      * @param nodes
297      *            the nodes
298      */
299     private void loadDecoratorMappers(NodeList nodes) {
300         clearDecoratorMappers();
301         Properties emptyProps = new Properties();
302 
303         pushDecoratorMapper("com.opensymphony.module.sitemesh.mapper.NullDecoratorMapper", emptyProps);
304 
305         // note, this works from the bottom node up.
306         for (int i = nodes.getLength() - 1; i > 0; i--) {
307             if (nodes.item(i) instanceof Element) {
308                 Element curr = (Element) nodes.item(i);
309                 if ("mapper".equalsIgnoreCase(curr.getTagName())) {
310                     String className = curr.getAttribute("class");
311                     Properties props = new Properties();
312                     // build properties from <param> tags.
313                     NodeList children = curr.getChildNodes();
314                     for (int j = 0; j < children.getLength(); j++) {
315                         if (children.item(j) instanceof Element) {
316                             Element currC = (Element) children.item(j);
317                             if ("param".equalsIgnoreCase(currC.getTagName())) {
318                                 String value = currC.getAttribute("value");
319                                 props.put(currC.getAttribute("name"), replaceProperties(value));
320                             }
321                         }
322                     }
323                     // add mapper
324                     pushDecoratorMapper(className, props);
325                 }
326             }
327         }
328 
329         pushDecoratorMapper("com.opensymphony.module.sitemesh.mapper.InlineDecoratorMapper", emptyProps);
330     }

332     /**
333      * Reads in all the url patterns to exclude from decoration.
334      *
335      * @param nodes
336      *            the nodes
337      */
338     private void loadExcludeUrls(NodeList nodes) {
339         clearExcludeUrls();
340         for (int i = 0; i < nodes.getLength(); i++) {
341             if (nodes.item(i) instanceof Element) {
342                 Element p = (Element) nodes.item(i);
343                 if ("pattern".equalsIgnoreCase(p.getTagName()) || "url-pattern".equalsIgnoreCase(p.getTagName())) {
344                     Text patternText = (Text) p.getFirstChild();
345                     if (patternText != null) {
346                         String pattern = patternText.getData().trim();
347                         if (pattern != null) {
348                             addExcludeUrl(pattern);
349                         }
350                     }
351                 }
352             }
353         }
354     }

356     /**
357      * Check if configuration file has been modified, and if so reload it.
358      */
359     @Override
360     public void refresh() {
361         long time = System.currentTimeMillis();
362         if (time - configLastCheck < configCheckMillis) {
363             return;
364         }
365         configLastCheck = time;
366 
367         if (configFile != null && configLastModified != configFile.lastModified()) {
368             loadConfig();
369         }
370     }

372     /**
373      * Replaces any properties that appear in the supplied string with their actual values.
374      *
375      * @param str
376      *            the string to replace the properties in
377      *
378      * @return the same string but with any properties expanded out to their actual values
379      */
380     private String replaceProperties(String str) {
381         int idx;
382         for (Map.Entry<Object, Object> entry : configProps.entrySet()) {
383             String key = (String) entry.getKey();
384             while ((idx = str.indexOf(key)) >= 0) {
385                 StringBuilder buf = new StringBuilder(100);
386                 buf.append(str.substring(0, idx));
387                 buf.append(entry.getValue());
388                 buf.append(str.substring(idx + key.length()));
389                 str = buf.toString();
390             }
391         }
392         return str;
393     }
