21  package com.opensymphony.module.sitemesh.freemarker;
22  
23  import com.opensymphony.module.sitemesh.HTMLPage;
24  import com.opensymphony.module.sitemesh.RequestConstants;
25  
26  import java.io.IOException;
27  import java.io.StringWriter;
28  
29  import javax.servlet.ServletException;
30  import javax.servlet.http.HttpServletRequest;
31  import javax.servlet.http.HttpServletResponse;
32  
33  import freemarker.ext.servlet.FreemarkerServlet;
34  import freemarker.template.SimpleHash;
35  import freemarker.template.Template;
36  import freemarker.template.TemplateModel;
37  
38  /**
39   * Servlet that allows Freemarker templates to be used as decorators.
40   *
41   * @author <a href="mailto:richard.hallier@freesbee.fr">Richard HALLIER</a>
42   */
43  public class FreemarkerDecoratorServlet extends FreemarkerServlet {
44  
45      /** The Constant serialVersionUID. */
46      private static final long serialVersionUID = 1L;
47  
48      @Override
49      protected boolean preTemplateProcess(HttpServletRequest request, HttpServletResponse response, Template template,
50              TemplateModel templateModel) throws ServletException, IOException {
51          boolean result = super.preTemplateProcess(request, response, template, templateModel);
52  
53          SimpleHash hash = (SimpleHash) templateModel;
54  
55          HTMLPage htmlPage = (HTMLPage) request.getAttribute(RequestConstants.PAGE);
56  
57          String title, body, head;
58  
59          if (htmlPage == null) {
60              title = "No Title";
61              body = "No Body";
62              head = "<!-- No head -->";
63          } else {
64              title = htmlPage.getTitle();
65  
66              StringWriter buffer = new StringWriter();
67              htmlPage.writeBody(buffer);
68              body = buffer.toString();
69  
70              buffer = new StringWriter();
71              htmlPage.writeHead(buffer);
72              head = buffer.toString();
73  
74              hash.put("page", htmlPage);
75          }
76  
77          hash.put("title", title);
78          hash.put("body", body);
79          hash.put("head", head);
80          hash.put("base", request.getContextPath());
81  
82          /*
83           * Factory factory = Factory.getInstance(new Config(getServletConfig())); Decorator decorator =
84           * factory.getDecoratorMapper().getDecorator(request, htmlPage); -> decorator.getPage()
85           */
86  
87          return result;
88      }
89  }
