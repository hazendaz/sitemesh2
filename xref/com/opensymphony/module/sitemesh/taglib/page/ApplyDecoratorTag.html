55  /**
56   * This tag inserts an external resource as a panel into the current Page.
57   * <p>
58   * The page attribute should point to the panel resource which should expose an entire page (e.g. another JSP file
59   * producing HTML). This attribute can be relative to the page it is being called from or an absolute path from the
60   * context-root.
61   * <p>
62   * <strong>OR</strong>
63   * <p>
64   * If the page attribute is not specified, the body content is parsed into the
65   * {@link com.opensymphony.module.sitemesh.Page} object and has the {@link com.opensymphony.module.sitemesh.Decorator}
66   * applied.
67   * <p>
68   * The (optional) decorator attribute is the name of the {@link com.opensymphony.module.sitemesh.Decorator} to apply to
69   * the included page. Note that the implementation of {@link com.opensymphony.module.sitemesh.DecoratorMapper} can
70   * overide this.
71   *
72   * @author <a href="mailto:joe@truemesh.com">Joe Walnes</a>
73   */
74  public class ApplyDecoratorTag extends BodyTagSupport implements RequestConstants {

194     @Override
195     public int doStartTag() {
196         if (config == null) {
197             // set context if not already set
198             config = new Config(pageContext.getServletConfig());
199             factory = Factory.getInstance(config);
200             decoratorMapper = factory.getDecoratorMapper();
201         }
202         // return page == null ? EVAL_BODY_BUFFERED : SKIP_BODY;
203         return EVAL_BODY_BUFFERED;
204     }
205 
206     /** Ensure that external page contents are included in bodycontent. */
207     @Override
208     public int doAfterBody() throws JspException {
209         return SKIP_BODY;
210     }
211 
212     /** Standard taglib method: apply decorator to page. */
213     @Override
214     public int doEndTag() throws JspException {
215         try {
216             // if composite decorator, remember last page
217             Page oldPage = (Page) pageContext.getRequest().getAttribute(PAGE);
218 
219             // parse bodycontent into Page object
220             PageParser parser = getParserSelector().getPageParser(contentType != null ? contentType : "text/html");
221             Page pageObj;
222 
223             if (page == null) {
224                 // inline content
225                 if (bodyContent != null) {
226                     // Would be nice if we could do our own buffering...
227                     SitemeshBufferWriter sitemeshWriter = new SitemeshBufferWriter();
228                     bodyContent.writeOut(sitemeshWriter);
229                     pageObj = parser.parse(sitemeshWriter.getSitemeshBuffer());
230                 } else {
231                     pageObj = parser.parse(new DefaultSitemeshBuffer(new char[] {}));
232                 }
233             } else if (page.startsWith("http://") || page.startsWith("https://")) {
234                 try {
235                     URL url = new URL(page);
236                     URLConnection urlConn = url.openConnection();
237                     urlConn.setUseCaches(true);
238 
239                     try (BufferedReader in = new BufferedReader(
240                             new InputStreamReader(urlConn.getInputStream(), Charset.forName(encoding)));
241                             SitemeshBufferWriter sitemeshWriter = new SitemeshBufferWriter()) {
242                         char[] buf = new char[1000];
243                         for (;;) {
244                             int moved = in.read(buf);
245                             if (moved < 0) {
246                                 break;
247                             }
248                             sitemeshWriter.write(buf, 0, moved);
249                         }
250                         pageObj = parser.parse(sitemeshWriter.getSitemeshBuffer());
251                     }
252                 } catch (IOException e) {
253                     throw new JspException(e);
254                 }
255             } else {
256                 // external content
257                 String fullPath = page;
258                 if (fullPath.length() > 0 && fullPath.charAt(0) != '/') {
259                     HttpServletRequest request = (HttpServletRequest) pageContext.getRequest();
260 
261                     // find absolute path if relative supplied
262                     String thisPath = request.getServletPath();
263 
264                     // check if it did not return null (could occur when the servlet container
265                     // does not use a servlet to serve the requested resouce)
266                     if (thisPath == null) {
267                         String requestURI = request.getRequestURI();
268                         if (request.getPathInfo() != null) {
269                             // strip the pathInfo from the requestURI
270                             thisPath = requestURI.substring(0, requestURI.indexOf(request.getPathInfo()));
271                         } else {
272                             thisPath = requestURI;
273                         }
274                     }
275 
276                     fullPath = thisPath.substring(0, thisPath.lastIndexOf('/') + 1) + fullPath;
277                     int dotdot;
278                     while ((dotdot = fullPath.indexOf("..")) > -1) {
279                         int prevSlash = fullPath.lastIndexOf('/', dotdot - 2);
280                         fullPath = fullPath.substring(0, prevSlash) + fullPath.substring(dotdot + 2);
281                     }
282                 }
283 
284                 // include page using filter response
285                 RequestDispatcher rd = pageContext.getServletContext().getRequestDispatcher(fullPath);
286                 PageRequestWrapper pageRequest = new PageRequestWrapper((HttpServletRequest) pageContext.getRequest());
287                 PageResponseWrapper pageResponse = new PageResponseWrapper(
288                         (HttpServletResponse) pageContext.getResponse(), factory);
289 
290                 StringBuilder sb = new StringBuilder(contentType != null ? contentType : "text/html");
291                 if (encoding != null) {
292                     sb.append(";charset=").append(encoding);
293                 }
294                 pageResponse.setContentType(sb.toString());
295 
296                 // if rd == null, then the panel was not found, but this isn't correct, so we need to spit out
297                 // something appropriate. What this is, well...I don't know yet.
298                 if (rd == null) {
299                     throw new ApplyDecoratorException(
300                             "The specified resource in applyDecorator tag (" + fullPath + ") was not found.");
301                 }
302                 rd.include(pageRequest, pageResponse);
303                 pageObj = pageResponse.getPage();
304             }
305 
306             // If pageObj == null, then the panel source had some weird error in
307             // it. Stop writing bugs like this. They're ugly and they make you smell funny.
308             if (pageObj == null) {
309                 throw new ApplyDecoratorException(page + " did not create a valid page to decorate.");
310             }
311 
312             // add extra params to Page
313             Iterator<String> paramKeys = params.keySet().iterator();
314             while (paramKeys.hasNext()) {
315                 String k = (String) paramKeys.next();
316                 String v = (String) params.get(k);
317                 pageObj.addProperty(k, v);
318             }
319 
320             // get decorator
321             if (decorator == null) {
322                 decorator = "";
323             }
324             pageObj.setRequest((HttpServletRequest) pageContext.getRequest());
325             pageContext.getRequest().setAttribute(DECORATOR, decorator);
326             Decorator d = decoratorMapper.getDecorator((HttpServletRequest) pageContext.getRequest(), pageObj);
327             pageContext.getRequest().removeAttribute(DECORATOR);
328 
329             // apply decorator
330             if ((d == null) || (d.getPage() == null)) {
331                 throw new JspException("Cannot locate inline Decorator: " + decorator);
332             }
333             pageContext.getRequest().setAttribute(PAGE, pageObj);
334             pageContext.include(d.getPage());
335 
336             // clean up
337             pageContext.getRequest().setAttribute(PAGE, oldPage);
338             params.clear(); // params need to be cleared between invocations - SIM-191
339         } catch (IOException | ServletException e) {
340             throw new JspException(e);
341         } catch (ApplyDecoratorException e) {
342             try {
343                 pageContext.getOut().println(e.getMessage());
344             } catch (IOException ioe) {
345                 System.err.println("IOException thrown in applyDecorator tag: " + e.toString());
346             }
347         }
348         return EVAL_PAGE;
349     }
