14  package com.opensymphony.module.sitemesh.taglib.decorator;
15  
16  import com.opensymphony.module.sitemesh.HTMLPage;
17  import com.opensymphony.module.sitemesh.taglib.AbstractTag;
18  
19  import java.io.IOException;
20  
21  import javax.servlet.jsp.JspException;
22  
23  /**
24   * Write a HTMLPage div to out.
25   *
26   * @see HTMLPage#writeHead(java.io.Writer)
27   */
28  public class DivTag extends AbstractTag {
29  
30      /** The Constant serialVersionUID. */
31      private static final long serialVersionUID = 1L;
32  
33      /** The div id. */
34      protected String divId;
35  
36      @Override
37      public String getId() {
38          return divId;
39      }
40  
41      @Override
42      public void setId(String divId) {
43          this.divId = divId;
44      }
45  
46      @Override
47      public final int doEndTag() throws JspException {
48          try {
49              String divBody = getPage().getProperty("div." + divId);
50              if (divBody != null) {
51                  getOut().write(divBody.substring(divBody.indexOf('>') + 1, divBody.lastIndexOf('<')));
52              }
53          } catch (IOException e) {
54              throw new JspException("Error writing head element: " + e.toString(), e);
55          }
56          return EVAL_PAGE;
57      }
58  }
