14  package com.opensymphony.sitemesh.compatability;
15  
16  import com.opensymphony.module.sitemesh.DecoratorMapper;
17  import com.opensymphony.sitemesh.Content;
18  import com.opensymphony.sitemesh.Decorator;
19  import com.opensymphony.sitemesh.DecoratorSelector;
20  import com.opensymphony.sitemesh.SiteMeshContext;
21  import com.opensymphony.sitemesh.webapp.SiteMeshWebAppContext;
22  import com.opensymphony.sitemesh.webapp.decorator.NoDecorator;
23  
24  import javax.servlet.http.HttpServletRequest;
25  
26  /**
27   * Adapts a SiteMesh 2 {@link DecoratorMapper} to a SiteMesh 2 {@link DecoratorSelector}.
28   *
29   * @author Joe Walnes
30   *
31   * @since SiteMesh 3
32   */
33  public class DecoratorMapper2DecoratorSelector implements DecoratorSelector {
34  
35      /** The decorator mapper. */
36      private final DecoratorMapper decoratorMapper;
37  
38      /**
39       * Instantiates a new decorator mapper 2 decorator selector.
40       *
41       * @param decoratorMapper
42       *            the decorator mapper
43       */
44      public DecoratorMapper2DecoratorSelector(DecoratorMapper decoratorMapper) {
45          this.decoratorMapper = decoratorMapper;
46      }
47  
48      @Override
49      public Decorator selectDecorator(Content content, SiteMeshContext context) {
50          SiteMeshWebAppContext webAppContext = (SiteMeshWebAppContext) context;
51          HttpServletRequest request = webAppContext.getRequest();
52          com.opensymphony.module.sitemesh.Decorator decorator = decoratorMapper.getDecorator(request,
53                  new Content2HTMLPage(content, request));
54          if (decorator == null || decorator.getPage() == null) {
55              return new NoDecorator();
56          }
57          return new OldDecorator2NewDecorator(decorator);
58      }
59  
60  }
