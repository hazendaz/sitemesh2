14  package com.opensymphony.sitemesh.compatability;
15  
16  import com.opensymphony.module.sitemesh.Factory;
17  import com.opensymphony.module.sitemesh.HTMLPage;
18  import com.opensymphony.module.sitemesh.Page;
19  import com.opensymphony.module.sitemesh.PageParser;
20  import com.opensymphony.module.sitemesh.SitemeshBuffer;
21  import com.opensymphony.module.sitemesh.filter.HttpContentType;
22  import com.opensymphony.sitemesh.Content;
23  import com.opensymphony.sitemesh.ContentProcessor;
24  import com.opensymphony.sitemesh.SiteMeshContext;
25  import com.opensymphony.sitemesh.webapp.SiteMeshWebAppContext;
26  
27  import jakarta.servlet.http.HttpServletRequest;
28  
29  import java.io.IOException;
30  
31  /**
32   * Adapts a SiteMesh 2 {@link PageParser} to a SiteMesh 3 {@link ContentProcessor}.
33   *
34   * @author Joe Walnes
35   *
36   * @since SiteMesh 3
37   */
38  public class PageParser2ContentProcessor implements ContentProcessor {
39  
40      /** The factory. */
41      private final Factory factory;
42  
43      /**
44       * Instantiates a new page parser 2 content processor.
45       *
46       * @param factory
47       *            the factory
48       */
49      public PageParser2ContentProcessor(Factory factory) {
50          this.factory = factory;
51      }
52  
53      /**
54       * Handles.
55       *
56       * @param context
57       *            the context
58       *
59       * @return true, if successful
60       */
61      @Override
62      public boolean handles(SiteMeshContext context) {
63          SiteMeshWebAppContext webAppContext = (SiteMeshWebAppContext) context;
64          return !factory.isPathExcluded(extractRequestPath(webAppContext.getRequest()));
65      }
66  
67      /**
68       * Extract request path.
69       *
70       * @param request
71       *            the request
72       *
73       * @return the string
74       */
75      private String extractRequestPath(HttpServletRequest request) {
76          String servletPath = request.getServletPath();
77          String pathInfo = request.getPathInfo();
78          String query = request.getQueryString();
79          return (servletPath == null ? "" : servletPath) + (pathInfo == null ? "" : pathInfo)
80                  + (query == null ? "" : "?" + query);
81      }
82  
83      @Override
84      public boolean handles(String contentType) {
85          return factory.shouldParsePage(contentType);
86      }
87  
88      /**
89       * Builds the.
90       *
91       * @param buffer
92       *            the buffer
93       * @param context
94       *            the context
95       *
96       * @return the content
97       *
98       * @throws IOException
99       *             Signals that an I/O exception has occurred.
100      */
101     @Override
102     public Content build(SitemeshBuffer buffer, SiteMeshContext context) throws IOException {
103         HttpContentType httpContentType = new HttpContentType(context.getContentType());
104         PageParser pageParser = factory.getPageParser(httpContentType.getType());
105         Page page = pageParser.parse(buffer);
106         return new HTMLPage2Content((HTMLPage) page);
107     }
108 
109 }
