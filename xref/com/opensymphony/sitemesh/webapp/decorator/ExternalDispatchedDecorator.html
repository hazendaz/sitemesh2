14  package com.opensymphony.sitemesh.webapp.decorator;
15  
16  import javax.servlet.ServletContext;
17  
18  /**
19   * Decorator that dispatches to another path in A DIFFERENT WEB-APP in the same Servlet Container (such as a JSP or path
20   * mapped to a Servlet).
21   * <p>
22   * The Content and SiteMeshContext objects are passed to the decorator using the HttpServletRequest attributes
23   * {@link #CONTENT_KEY} and {@link #CONTEXT_KEY}.
24   * <p>
25   * To dispatch to a decorator in the same web-app, use {@link DispatchedDecorator}.
26   *
27   * @author Joe Walnes
28   *
29   * @since SiteMesh 3.0
30   */
31  public class ExternalDispatchedDecorator extends DispatchedDecorator {
32  
33      /** The web app. */
34      private final String webApp;
35  
36      /**
37       * Instantiates a new external dispatched decorator.
38       *
39       * @param path
40       *            the path
41       * @param webApp
42       *            the web app
43       */
44      public ExternalDispatchedDecorator(String path, String webApp) {
45          super(path);
46          this.webApp = webApp;
47      }
48  
49      @Override
50      protected ServletContext locateWebApp(ServletContext context) {
51          ServletContext externalContext = context.getContext(webApp);
52          if (externalContext != null) {
53              return externalContext;
54          }
55          // in a security conscious environment, the servlet container
56          // may return null for a given URL
57          throw new SecurityException("Cannot obtain ServletContext for web-app: " + webApp);
58      }
59  
60  }
