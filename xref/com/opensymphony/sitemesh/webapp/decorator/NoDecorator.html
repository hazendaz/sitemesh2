14  package com.opensymphony.sitemesh.webapp.decorator;
15  
16  import com.opensymphony.sitemesh.Content;
17  import com.opensymphony.sitemesh.webapp.SiteMeshWebAppContext;
18  
19  import java.io.IOException;
20  import java.io.PrintWriter;
21  
22  import javax.servlet.ServletContext;
23  import javax.servlet.ServletException;
24  import javax.servlet.http.HttpServletRequest;
25  import javax.servlet.http.HttpServletResponse;
26  
27  /**
28   * If no decorator is to be applied to a page, this will ensure the original content gets written out.
29   *
30   * @author Joe Walnes
31   *
32   * @since SiteMesh 3.0
33   */
34  public class NoDecorator extends BaseWebAppDecorator {
35  
36      /**
37       * Render.
38       *
39       * @param content
40       *            the content
41       * @param request
42       *            the request
43       * @param response
44       *            the response
45       * @param servletContext
46       *            the servlet context
47       * @param webAppContext
48       *            the web app context
49       *
50       * @throws IOException
51       *             Signals that an I/O exception has occurred.
52       * @throws ServletException
53       *             the servlet exception
54       */
55      @Override
56      protected void render(Content content, HttpServletRequest request, HttpServletResponse response,
57              ServletContext servletContext, SiteMeshWebAppContext webAppContext) throws IOException, ServletException {
58  
59          if (webAppContext.isUsingStream()) {
60              PrintWriter writer = new PrintWriter(response.getOutputStream());
61              content.writeOriginal(writer);
62              writer.flush(); // flush writer to underlying outputStream
63              response.getOutputStream().flush();
64          } else {
65              PrintWriter writer = response.getWriter();
66              content.writeOriginal(writer);
67              response.getWriter().flush();
68          }
69      }
70  
71  }
