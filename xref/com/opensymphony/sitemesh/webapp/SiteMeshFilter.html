37  /**
38   * Core Filter for integrating SiteMesh into a Java web application.
39   *
40   * @author Joe Walnes
41   * @author Scott Farquhar
42   *
43   * @since SiteMesh 3
44   */
45  public class SiteMeshFilter implements Filter {
46  
47      /** The filter config. */
48      private FilterConfig filterConfig;
49  
50      /** The container tweaks. */
51      private ContainerTweaks containerTweaks;
52  
53      /** The Constant ALREADY_APPLIED_KEY. */
54      private static final String ALREADY_APPLIED_KEY = "com.opensymphony.sitemesh.APPLIED_ONCE";
55  
56      @Override
57      public void init(FilterConfig filterConfig) {
58          this.filterConfig = filterConfig;
59          containerTweaks = new ContainerTweaks();
60      }
61  
62      @Override
63      public void destroy() {
64          filterConfig = null;
65          containerTweaks = null;
66      }
67  
68      /**
69       * Main method of the Filter.
70       * <p>
71       * Checks if the Filter has been applied this request. If not, parses the page and applies
72       * {@link com.opensymphony.module.sitemesh.Decorator} (if found).
73       */
74      @Override
75      public void doFilter(ServletRequest rq, ServletResponse rs, FilterChain chain)
76              throws IOException, ServletException {
77  
78          HttpServletRequest request = (HttpServletRequest) rq;
79          HttpServletResponse response = (HttpServletResponse) rs;
80          ServletContext servletContext = filterConfig.getServletContext();
81  
82          SiteMeshWebAppContext webAppContext = new SiteMeshWebAppContext(request, response, servletContext);
83  
84          ContentProcessor contentProcessor = initContentProcessor(webAppContext);
85          DecoratorSelector decoratorSelector = initDecoratorSelector(webAppContext);
86  
87          if (filterAlreadyAppliedForRequest(request) || !contentProcessor.handles(webAppContext)) {
88              // Optimization: If the content doesn't need to be processed, bypass SiteMesh.
89              chain.doFilter(request, response);
90              return;
91          }
92  
93          if (containerTweaks.shouldAutoCreateSession()) {
94              // Some containers (such as Tomcat 4) will not allow sessions to be created in the decorator.
95              // (i.e after the response has been committed).
96              request.getSession(true);
97          }
98  
99          try {
100 
101             Content content = obtainContent(contentProcessor, webAppContext, request, response, chain);
102 
103             if (content == null) {
104                 request.setAttribute(ALREADY_APPLIED_KEY, null);
105                 return;
106             }
107 
108             Decorator decorator = decoratorSelector.selectDecorator(content, webAppContext);
109             decorator.render(content, webAppContext);
110 
111         } catch (IllegalStateException e) {
112             // Some containers (such as WebLogic) throw an IllegalStateException when an error page is served.
113             // It may be ok to ignore this. However, for safety it is propegated if possible.
114             if (!containerTweaks.shouldIgnoreIllegalStateExceptionOnErrorPage()) {
115                 throw e;
116             }
117         } catch (RuntimeException e) {
118             if (containerTweaks.shouldLogUnhandledExceptions()) {
119                 // Some containers (such as Tomcat 4) swallow RuntimeExceptions in filters.
120                 servletContext.log("Unhandled exception occurred whilst decorating page", e);
121             }
122             throw e;
123         } catch (ServletException e) {
124             request.setAttribute(ALREADY_APPLIED_KEY, null);
125             throw e;
126         }
127 
128     }
129 
130     /**
131      * Inits the content processor.
132      *
133      * @param webAppContext
134      *            the web app context
135      *
136      * @return the content processor
137      */
138     protected ContentProcessor initContentProcessor(SiteMeshWebAppContext webAppContext) {
139         // TODO: Remove heavy coupling on horrible SM2 Factory
140         Factory factory = Factory.getInstance(new Config(filterConfig));
141         factory.refresh();
142         return new PageParser2ContentProcessor(factory);
143     }
144 
145     /**
146      * Inits the decorator selector.
147      *
148      * @param webAppContext
149      *            the web app context
150      *
151      * @return the decorator selector
152      */
153     protected DecoratorSelector initDecoratorSelector(SiteMeshWebAppContext webAppContext) {
154         // TODO: Remove heavy coupling on horrible SM2 Factory
155         Factory factory = Factory.getInstance(new Config(filterConfig));
156         factory.refresh();
157         return new DecoratorMapper2DecoratorSelector(factory.getDecoratorMapper());
158     }
159 
160     /**
161      * Continue in filter-chain, writing all content to buffer and parsing into returned
162      * {@link com.opensymphony.module.sitemesh.Page} object. If {@link com.opensymphony.module.sitemesh.Page} is not
163      * parseable, null is returned.
164      *
165      * @param contentProcessor
166      *            the content processor
167      * @param webAppContext
168      *            the web app context
169      * @param request
170      *            the request
171      * @param response
172      *            the response
173      * @param chain
174      *            the chain
175      *
176      * @return the content
177      *
178      * @throws IOException
179      *             Signals that an I/O exception has occurred.
180      * @throws ServletException
181      *             the servlet exception
182      */
183     private Content obtainContent(ContentProcessor contentProcessor, SiteMeshWebAppContext webAppContext,
184             HttpServletRequest request, HttpServletResponse response, FilterChain chain)
185             throws IOException, ServletException {
186 
187         ContentBufferingResponse contentBufferingResponse = new ContentBufferingResponse(response, contentProcessor,
188                 webAppContext);
189         chain.doFilter(request, contentBufferingResponse);
190         // TODO: check if another servlet or filter put a page object in the request
191         // Content result = request.getAttribute(PAGE);
192         // if (result == null) {
193         // // parse the page
194         // result = pageResponse.getPage();
195         // }
196         webAppContext.setUsingStream(contentBufferingResponse.isUsingStream());
197         return contentBufferingResponse.getContent();
198     }
199 
200     /**
201      * Filter already applied for request.
202      *
203      * @param request
204      *            the request
205      *
206      * @return true, if successful
207      */
208     private boolean filterAlreadyAppliedForRequest(HttpServletRequest request) {
209         if (request.getAttribute(ALREADY_APPLIED_KEY) == Boolean.TRUE) {
210             return true;
211         }
212         request.setAttribute(ALREADY_APPLIED_KEY, Boolean.TRUE);
213         return false;
214     }
